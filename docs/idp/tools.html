 28class Measure:
 29    r'''A finite, signed Borel measure on the real line.
 30
 31    `Measure` objects can be added to each other and multiplied on the
 32    left by a `float`. These operations return new `Measure` objects.
 33    When adding two instances, the `bandwidth` of the result is the
 34    maximum of the bandwidths of the summands.
 35
 36    '''
 37    def __init__(self, cdf, pdf, mean=None, totalMass=1):
 38        '''Constructs a `Measure` object from the given arguments.
 39
 40        '''
 41        self._cdf = cdf
 42        self._pdf = pdf
 43        self._mean = mean
 44        self._totalMass = totalMass
 45
 46    @property
 47    def cdf(self):
 48        r'''`func`: The distribution function of the measure. If $\mu$
 49        is the measure, then this is the function $t \mapsto
 50        \mu((-\infty, t])$.
 51        '''
 52        return self._cdf
 53
 54    @property
 55    def pdf(self):
 56        '''`func`: The density of the measure, which is the derivative
 57        of `cdf`. If `cdf` is not differentiable, then an approximate
 58        density should be provided.
 59        '''
 60        return self._pdf
 61
 62    @property
 63    def mean(self):
 64        r'''`float` or `None`: If the mean is stored in the `Measure`
 65        instance, then it should equal $\int_{-\infty}^\infty x \,
 66        dF(x)$, where $F$ is the distribution function. The mean of the
 67        sum of two `Measure` instances will be `None` if either mean is
 68        `None`.
 69        '''
 70        return self._mean
 71
 72    @property
 73    def totalMass(self):
 74        r'''`float`: The total mass of the measure. If $\mu$ is the
 75        measure, then this $\mu(\mathbb{R})$.
 76        '''
 77        return self._totalMass
 78
 79    @property
 80    def bandwidth(self):
 81        '''`float`: If `pdf` is an approximate density created with the
 82        `gaussian_kde` function from the `scipy.stats` module, then this
 83        is the bandwidth that was used to create it. Is 0 otherwise. See
 84        `IDPModel.rowDistLaw` for more details.
 85        '''
 86        try:
 87            return self.pdf.factor
 88        except AttributeError:
 89            return 0
 90
 91    def __rmul__(self, scalar):
 92        def pdf(t):
 93            return scalar * self.pdf(t)
 94        pdf.factor = self.bandwidth
 95        mean = (
 96            None if self._mean is None
 97            else scalar * self._mean
 98        )
 99        return Measure(
100            lambda t: scalar * self._cdf(t),
101            pdf,
102            mean,
103            scalar * self._totalMass
104        )
105
106    def __add__(self, other):
107        def pdf(t):
108            return self._pdf(t) + other.pdf(t)
109        pdf.factor = max(self.bandwidth, other.bandwidth)
110        mean = (
111            None if self._mean is None or other.mean is None
112            else self._mean + other.mean
113        )
114        return Measure(
115            lambda t: self._cdf(t) + other.cdf(t),
116            pdf,
117            mean,
118            self._totalMass + other.totalMass
119        )

121class IDPModel:
122    r'''An IDP model on the state space $S = \\{0, ..., L - 1\\}$.
123
124    The IDP model consists of both an IDP and some observed data. See
125    Section 9 of the Paper for details. The bijection $\sigma$ that
126    appears in the Paper is taken here to be the identity.
127
128    '''
129
130    def __init__(self, colConc, rowConc, baseMeas, rowCounts):
131        '''Uses the given arguments to set the corresponding properties.
132        The `baseMeas` and `rowCounts` arguments should be numpy arrays
133        of the same shape as the corresponding properties.
134
135        '''
136        self._idp = {
137            'colConc': colConc,
138            'rowConc': rowConc,
139            'baseMeas': baseMeas
140        }
141        self._rowCounts = rowCounts
142
143        rowConc = self._idp['rowConc']
144        baseMeas = self._idp['baseMeas']
145        denom = betaFuncMult(rowConc * baseMeas)
146        self._likelihoods = [
147            betaFuncMult(rowConc * baseMeas + row) / denom
148            for row in rowCounts
149        ]
150
151        self._scale = 0
152        self._weightedSims = []
153
154    @property
155    def colConc(self):
156        r'''`float`: The column concentration parameter. Should be
157        positive. Denoted by $\kappa$ in Section 9 of the Paper.
158        '''
159        return self._idp['colConc']
160
161    @property
162    def rowConc(self):
163        r'''`float`: The row concentration parameter. Should be
164        positive. Denoted by $\varepsilon$ in Section 9 of the Paper.
165        '''
166        return self._idp['rowConc']
167
168    @property
169    def baseMeas(self):
170        r'''`tuple`(`float`): The base probability vector. Should be a
171        sequence of positive numbers that sum to 1. Denoted by
172        $\boldsymbol{p} = (p_0, \ldots, p_{L - 1})$ in Section 9 of the
173        Paper, where $p_\ell = \varrho(\\{\ell\\})$.
174        '''
175        return tuple(self._idp['baseMeas'])
176
177    @property
178    def rowCounts(self):
179        r'''`tuple`(`tuple`(`int`)): The row counts for the observed
180        data. If `row` is $m - 1$ and `state` is $\ell$, then
181        `rowCounts`[`row`][`state`] is denoted by $\overline{y}_{m\ell}$
182        in Section 9 of the Paper. It is the number of observed
183        instances of `state` in the given row of data.
184        '''
185        return tuple(tuple(row) for row in self._rowCounts)
186
187    @property
188    def likelihoods(self):
189        r'''`tuple`(`float`): The prior likelihoods. If `row` equals
190        $m - 1$, then `likelihoods`[`row`] is denoted by
191        $\varrho_N(A_m)$ in Section 9 of the Paper. It is the prior
192        probability of seeing the row counts for the given row.
193        '''
194        return tuple(self._likelihoods)
195
196    @property
197    def scale(self):
198        r'''`float`: The scale factor to use for weighted simulations.
199        Denoted by $\log c$ in Section 9 of the Paper. Defaults to 0.
200        Altering this value runs the `clearWeightedSims` method.
201        '''
202        return self._scale
203
204    @scale.setter
205    def scale(self, value):
206        self._scale = value
207        self.clearWeightedSims()
208
209    @property
210    def weightedSims(self):
211        r'''`tuple`(`WeightedSim`): A tuple of distinct (and therefore
212        independent) `WeightedSim` instances. `weightedSims`[k] is
213        denoted by $(t^k, \boldsymbol{\theta}_M^{*, k})$ in Section 9 of
214        the Paper, and has total weight $V_k$. This property is empty if
215        the `addWeightedSims` method has never been run.
216        '''
217        return tuple(self._weightedSims)
218
219    @property
220    def ess(self):
221        '''The effective sample size, approximated with the sample
222        variance. Denoted by $K_e''$ in Section 7.2 of the Paper.
223        '''
224        numSims = len(self._weightedSims)
225        if numSims == 0:
226            return 0
227        essPop = (
228            sum(sim.totalWeight      for sim in self._weightedSims) ** 2 /
229            sum(sim.totalWeight ** 2 for sim in self._weightedSims)
230        )
231        return essPop * (numSims - 1) / (numSims - essPop / numSims)
232
233    def addWeightedSims(self, num):
234        '''Creates new `WeightedSim` instances and adds them to the
235        `weightedSims` property.
236
237        **Arguments:**
238        
239        * **num** (`int`): The number of distinct `WeightedSim`
240            instances to create.
241
242        '''
243        self._weightedSims += [WeightedSim(self) for _ in range(num)]
244
245    def clearWeightedSims(self):
246        '''Deletes all weighted simulations contained in the
247        `weightedSims` property.
248
249        '''
250        self._weightedSims = []

252    def prior(self, mapping, numSamples=10000, bandwidth=None):
253        r'''Returns the pushforward of the Dirichlet distribution by the
254        given function. Used in the implementation of (8.24), (8.26),
255        (9.6), and the final displayed equation in Section 9 of the
256        paper. If an integer $\ell$ is passed to `mapping`, then the
257        function $\nu \mapsto \nu(\\{\ell\\})$ is used. If `mapping` is
258        a function, then the pdf of the pushforward is estimated by
259        sampling and Gaussian kernel density estimation (KDE). See
260        `rowDistLaw` for details about KDE.
261
262        **Arguments:**
263
264        * **mapping** (`func` or `int`): The function to use in the
265            pushforward.
266        * **numSamples** (`int`): The number of samples to use when
267            estimating the pdf.
268        * **bandwidth** (`float`): The bandwidth to use in the KDE.
269
270        **Returns:**
271
272        * `Measure`: The distribution of $\mathcal{D}(\varepsilon
273            \varrho) \circ \Phi^{-1}$, where $\varepsilon$ is `colConc`,
274            $\varrho$ is `baseMeas`, and $\Phi$ is `mapping`.
275
276        '''
277        if not callable(mapping):
278            prior = beta(
279                self._idp['rowConc'] * self._idp['baseMeas'][mapping],
280                self._idp['rowConc'] * (1 - self._idp['baseMeas'][mapping])
281            )
282            return Measure(prior.cdf, prior.pdf, prior.mean())
283        rawSamples = dirichlet.rvs(
284            self._idp['rowConc'] * self._idp['baseMeas'],
285            numSamples
286        )
287        samples = [mapping(sample) for sample in rawSamples]
288        pdf = gaussian_kde(samples, bandwidth)
289        return Measure(
290            ecdf(samples).cdf.evaluate,
291            pdf,
292            np.average(samples),
293            pdf.factor
294        )
295
296    def rowDistLaw(self, rowNum=None, mapping=1, bandwidth=None):
297        r'''Returns the law (or distribution) of a real-valued function
298        of a row distribution. If `mapping` is $\Phi$ and `rowNum` is
299        $m - 1$, then this method returns the law of the real-valued
300        random variable, $\Phi(\mu_m)$. If $M$ denotes the length of
301        `rowCounts`, then this method implements (8.10) in the Paper
302        when $m \le M$ and it implements (8.24) when $m > M$.
303
304        The law of $\Phi(\mu_m)$ is a Borel probability measure on the
305        real line. This method therefore returns a `Measure` object.
306        However, the law of $\Phi(\mu_m)$ has a discrete component, so
307        it does not have a density. The `Measure.pdf` attribute of the
308        returned object is an approximate density, created with Gaussian
309        kernel density estimation (KDE). More specifically, it treats a
310        point mass $\delta_x$ as having a Gaussian density with mean $x$
311        and standard deviation $h$, where $h$ is the "bandwidth" of the
312        estimate.
313
314        **Arguments:**
315
316        * **rowNum** (`int` or `None`): Passing `None` to this argument
317            is the same as passing the length of `rowCounts`.
318        * **mapping** (`func` or `int`): The function to apply to the
319            row distribution. Should be continuous and bounded. If an
320            integer $\ell$ is passed to `mapping`, then the function
321            $\mu \mapsto \mu(\\{\ell\\})$ is used.
322        * **bandwidth** (`float`): The bandwidth to use in the Gaussian
323            KDE. If omitted, the bandwidth will be calculated using
324            Scott's Rule. The calculated value can be retrieved by
325            accessing the `Measure.bandwidth` property of the returned
326            object. This argument is passed directly to the
327            `gaussian_kde` function in the `scipy.stats` module. See
328            that documentation for further details.
329
330        **Returns:**
331
332        * `Measure`: The distribution of $\Phi(\mu_m)$, which is a Borel
333            probability measure on the real line.
334
335        **Raises:**
336
337        * **ValueError**: If the `weightedSims` property is empty.
338
339        '''
340        if not self._weightedSims:
341            raise ValueError('No weighted simulations found')
342
343        func = mapping if callable(mapping) else lambda meas: meas[mapping]
344        numRows = len(self._rowCounts)
345        rowNum = numRows if rowNum is None else rowNum
346
347        # Loop through all rows of the weighted sims unless a valid row
348        # number is given.
349        if rowNum < numRows:
350            loopLength = 1
351            loopRange = [rowNum]
352        else:
353            loopLength = numRows
354            loopRange = range(loopLength)
355
356        # Created the list of weighted samples. If looping through all
357        # rows, the `sample` list is ordered as
358        # \[
359        #   u_1^{*, 1}, \ldots, u_M^{*, 1},
360        #   u_1^{*, 2}, \ldots, u_M^{*, 2},
361        #   \ldots,
362        #   u_1^{*, K}, \ldots, u_M^{*, K}
363        # \]
364        samples = np.array([
365            func(weightedSim.rowDistSims[m])
366            for weightedSim in self._weightedSims # outer loop
367            for m in loopRange  # inner loop
368        ])
369        weights = np.concatenate([
370            [weightedSim.totalWeight] * loopLength
371            for weightedSim in self._weightedSims
372        ])
373
374        # Create the sample law
375        sampleCDF = ECDFDiscrete(samples, weights)
376        samplePDF = gaussian_kde(samples, bandwidth, weights)
377        sampleLaw = Measure(
378            sampleCDF,
379            samplePDF,
380            np.average(samples, weights=weights),
381            samplePDF.factor
382        )
383
384        # If looking at a valid row number, return only the sample law
385        # as in (8.10). Otherwise, return a mixture as in (8.24).
386        if rowNum < numRows:
387            return sampleLaw
388        colConc = self._idp['colConc']
389        return (
390            (colConc / (colConc + numRows)) * self.prior(mapping) +
391            (numRows / (colConc + numRows)) * sampleLaw
392        )

394class WeightedSim:
395    r'''A weighted simulation for an `IDPModel` instance.
396
397    In the Paper, a weighted simulation is determined by the pair $(t,
398    \boldsymbol{\theta}_M^*)$ and has total weight $V$.
399
400    '''
401
402    def __init__(self, idpModel):
403        r'''Constructs a weighted simulation for the given IDP model by
404        computing the `rowWeights`, `rowDistSims`, and `totalWeight`
405        properties.
406
407        **Arguments:**
408
409        * **idpModel** (`IDPModel`): The IDP model for which the
410            weighted simulation is being created.
411
412        **Raises:**
413
414        * **ValueError**: If the computed value for the `totalWeight`
415            property is so small or so large that it evaluates to `0.0`
416            or `np.inf`.
417
418        '''
419        self._rowWeights = []
420        self._rowDistSims = []
421        self._logWeight = 0 # the natural logarithm of the total weight
422
423        # If $L$ is the number of states in the IDP, then the case
424        # $L = 2$ is separated from the case $2 < L < \infty$ to speed
425        # up code execution.
426        baseMeas = np.array(idpModel.baseMeas)
427        numStates = len(baseMeas)
428
429        # expand properties if row counts have been added
430        for m, counts in enumerate(idpModel.rowCounts):
431
432            # build `rowWeights[m]`
433            self._rowWeights.append([])
434            for i in range(m):
435                if numStates == 2:
436                    self._rowWeights[m].append(
437                        self._rowDistSims[i][0] ** counts[0] *
438                        self._rowDistSims[i][1] ** counts[1]
439                    )
440                else:
441                    weight = self._rowDistSims[i][0] ** counts[0]
442                    for state in range(1, numStates):
443                        weight *= self._rowDistSims[i][state] ** counts[state]
444                    self._rowWeights[m].append(weight)
445            self._rowWeights[m].append(
446                idpModel.colConc * idpModel.likelihoods[m]
447            )
448
449            # build `rowDistSims[m]`
450            choice = choices(range(m + 1), weights=self._rowWeights[m])[0]
451            if choice < m:
452                self._rowDistSims.append(self._rowDistSims[choice])
453            else:
454                if numStates == 2:
455                    rowDistSim = beta.rvs(
456                        idpModel.rowConc * baseMeas[1] + counts[1],
457                        idpModel.rowConc * baseMeas[0] + counts[0]
458                    )
459                    self._rowDistSims.append([1 - rowDistSim, rowDistSim])
460                else:
461                    self._rowDistSims.append(
462                        # vector sum fails for lists, works for ndarray
463                        dirichlet.rvs(
464                            idpModel.rowConc * baseMeas + counts
465                        )[0]
466                    )
467
468            # build `totalWeight`
469            #
470            # The m-th iteration of the `for` loop multiplies the
471            # previous value of `totalWeight` by the (m + 1)-th factor
472            # in the definition of $V$ in Section 9 of the Paper.
473            self._logWeight += (
474                idpModel.scale +
475                np.log(m + 1) -
476                np.log(idpModel.colConc + m) +
477                np.log(sum(self._rowWeights[m]))
478            )
479
480        # check that total weight is not out of bounds
481        if np.exp(self._logWeight) in (0, np.inf):
482            adj = -self._logWeight/len(idpModel.rowCounts)
483            raise ValueError(
484                'Simulation weight out of bounds.\n' +
485                f'Try adjusting scale factor by {adj}'
486            )
487
488    @property
489    def rowWeights(self):
490        '''`tuple`(`tuple`(`float`)): The triangular array of row
491        weights for the weighted simulation. Denoted by $t$ in Section 9
492        of the Paper.
493        '''
494        return tuple(tuple(row) for row in self._rowWeights)
495
496    @property
497    def rowDistSims(self):
498        r'''`tuple`(`tuple`(`float`)): A tuple of simulated row
499        distributions. If `row` is $m - 1$ and `state` is $\ell$, then
500        `rowDistSims`[`row`][`state`] is denoted by $\theta_{m\ell}^*
501        = u_m^*(\\{\ell\\})$ in Section 9 of the Paper.
502        '''
503        return tuple(self._rowDistSims)
504
505    @property
506    def totalWeight(self):
507        '''`float`: The total weight of the simulation. Denoted by $V$
508        in Section 9 of the Paper.
509        '''
510        return np.exp(self._logWeight)

358        # rows, the `sample` list is ordered as

405        computing the `rowWeights`, `rowDistSims`, and `totalWeight`
406        properties.
407
408        **Arguments:**
409
410        * **idpModel** (`IDPModel`): The IDP model for which the
411            weighted simulation is being created.
412
413        **Raises:**
414
415        * **ValueError**: If the computed value for the `totalWeight`
416            property is so small or so large that it evaluates to `0.0`
417            or `np.inf`.
418
419        '''
420        self._rowWeights = []
421        self._rowDistSims = []
422        self._logWeight = 0 # the natural logarithm of the total weight
423
424        # If $L$ is the number of states in the IDP, then the case
425        # $L = 2$ is separated from the case $2 < L < \infty$ to speed
426        # up code execution.
427        baseMeas = np.array(idpModel.baseMeas)
428        numStates = len(baseMeas)
429
430        # expand properties if row counts have been added
431        for m, counts in enumerate(idpModel.rowCounts):
432
433            # build `rowWeights[m]`
434            self._rowWeights.append([])
435            for i in range(m):
436                if numStates == 2:
437                    self._rowWeights[m].append(
438                        self._rowDistSims[i][0] ** counts[0] *
439                        self._rowDistSims[i][1] ** counts[1]
440                    )
441                else:
442                    weight = self._rowDistSims[i][0] ** counts[0]
443                    for state in range(1, numStates):
444                        weight *= self._rowDistSims[i][state] ** counts[state]
445                    self._rowWeights[m].append(weight)
446            self._rowWeights[m].append(
447                idpModel.colConc * idpModel.likelihoods[m]
448            )
449
450            # build `rowDistSims[m]`
451            choice = choices(range(m + 1), weights=self._rowWeights[m])[0]
452            if choice < m:
453                self._rowDistSims.append(self._rowDistSims[choice])
454            else:
455                if numStates == 2:
456                    rowDistSim = beta.rvs(
457                        idpModel.rowConc * baseMeas[1] + counts[1],
458                        idpModel.rowConc * baseMeas[0] + counts[0]
459                    )
460                    self._rowDistSims.append([1 - rowDistSim, rowDistSim])
461                else:
462                    self._rowDistSims.append(
463                        # vector sum fails for lists, works for ndarray
464                        dirichlet.rvs(
465                            idpModel.rowConc * baseMeas + counts
466                        )[0]
467                    )
468
469            # build `totalWeight`
470            #
471            # The m-th iteration of the `for` loop multiplies the
472            # previous value of `totalWeight` by the (m + 1)-th factor
473            # in the definition of $V$ in Section 9 of the Paper.
474            self._logWeight += (
475                idpModel.scale +
476                np.log(m + 1) -
477                np.log(idpModel.colConc + m) +
478                np.log(sum(self._rowWeights[m]))
479            )
480
481        # check that total weight is not out of bounds
482        if np.exp(self._logWeight) in (0, np.inf):
483            adj = -self._logWeight/len(idpModel.rowCounts)
484            raise ValueError(
485                'Simulation weight out of bounds.\n' +
486                f'Try adjusting scale factor by {adj}'
487            )
