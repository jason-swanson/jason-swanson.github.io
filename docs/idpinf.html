 11class DiscMeas:
 12    r'''A finite, signed measure on the nonnegative integers.
 13
 14    `DiscMeas` objects can be added to each other and multiplied on the
 15    left by a `float`. These operations return new `DiscMeas` objects.
 16
 17    '''
 18    def __init__(self, pmf, mean=None, totalMass=1):
 19        '''Constructs a `DiscMeas` object from the given arguments.
 20
 21        '''
 22        self._pmf = pmf
 23        self._mean = mean
 24        self._totalMass = totalMass
 25
 26    @property
 27    def pmf(self):
 28        r'''`func`: The mass function of the measure. If $\varrho$ is
 29        the measure, then this is the function $\ell \mapsto p_\ell$,
 30        where $p_\ell = \varrho(\\{\ell\\})$.
 31        '''
 32        return self._pmf
 33
 34    @property
 35    def mean(self):
 36        r'''`float` or `None`: If the mean is stored in the `DiscMeas`
 37        instance, then it should equal $\sum_{\ell = 0}^\infty \ell
 38        \cdot p_\ell$. The mean of the sum of two `DiscMeas` instances
 39        will be `None` if either mean is `None`.
 40        '''
 41        return self._mean
 42
 43    @property
 44    def totalMass(self):
 45        r'''`float`: The total mass of the measure. Is equal to
 46        $\sum_{\ell = 0}^\infty p_\ell$.
 47        '''
 48        return self._totalMass
 49
 50    def __rmul__(self, scalar):
 51        mean = (
 52            None if self._mean is None
 53            else scalar * self._mean
 54        )
 55        return DiscMeas(
 56            lambda t: scalar * self._pmf(t),
 57            mean,
 58            scalar * self._totalMass
 59        )
 60
 61    def __add__(self, other):
 62        mean = (
 63            None if self._mean is None or other.mean is None
 64            else self._mean + other.mean
 65        )
 66        return DiscMeas(
 67            lambda t: self._pmf(t) + other.pmf(t),
 68            mean,
 69            self._totalMass + other.totalMass
 70        )

 72class IDPModelBase:
 73    '''The base class that `IDPModelInf` inherits from.
 74
 75    Subclasses must override the `_computeLLH` method, which computes or
 76    recomputes, if necessary, the value of `likelihoods` and stores it
 77    in the `_likelihoods` attribute.
 78
 79    '''
 80
 81    def __init__(self, colConc, rowConc):
 82        '''Constructs an `IDPModelBase` object from the given arguments.
 83
 84        '''
 85        self._idp = {
 86            'colConc': colConc,
 87            'rowConc': rowConc
 88        }
 89        self._likelihoods = []
 90
 91    @property
 92    def colConc(self):
 93        r'''`float`: The column concentration parameter. Should be
 94        positive. Denoted by $\kappa$ in Section 9 of the Paper.
 95        '''
 96        return self._idp['colConc']
 97
 98    @property
 99    def rowConc(self):
100        r'''`float`: The row concentration parameter. Should be
101        positive. Denoted by $\varepsilon$ in Section 9 of the Paper.
102        '''
103        return self._idp['rowConc']
104
105    @property
106    def likelihoods(self):
107        r'''`tuple`(`float`): The prior likelihoods. If `row` equals
108        $m - 1$, then `likelihoods`[`row`] is denoted by
109        $\varrho_N(A_m)$ in Section 9 of the Paper. It is the prior
110        probability of seeing the row counts for the given row.
111        '''
112        self._computeLLH()
113        return tuple(self._likelihoods)
114
115    def _computeLLH(self):
116        raise NotImplementedError

123    def __init__(self, colConc, rowConc, baseMeas, data):
124        '''The constructor first sets the `colConc`, `rowConc`,
125        `baseMeas`, and `data` properties using the given arguments,
126        then builds the `rowCounts` property from the `data` argument.
127        The `data` argument can be a list of lists.
128
129        '''
130        super().__init__(colConc, rowConc)
131
132        self._idp['baseMeas'] = baseMeas
133
134        self._data = data
135        def pmf(rowCount, state):
136            return rowCount[state]
137        self._rowCounts = [
138            DiscMeas(partial(pmf, Counter(row)), fmean(row), len(row))
139            for row in data
140        ]
141
142    @property
143    def baseMeas(self):
144        r'''`DiscMeas`: The base probability measure. Denoted by
145        $p = \\{p_\ell\\}_{\ell = 0}^\infty$ in Section 9 of the
146        Paper, where $p_\ell = \varrho(\\{\ell\\})$.
147        '''
148        return self._idp['baseMeas']
149
150    @property
151    def data(self):
152        '''`tuple`(`tuple`(`int`)): The observed data. If `row` is
153        $m - 1$ and `col` is $n - 1$, then `data`[`row`][`col`] is
154        denoted by $y_{mn}$ in Section 9 of the Paper.
155        '''
156        return self._data
157
158    @property
159    def maxObserved(self):
160        '''`int`: The largest observed state in the `data` property.
161        Denoted by $L - 1$ in Section 9 of the Paper.
162        '''
163        return max(sum(self._data, []))
164
165    @property
166    def rowCounts(self):
167        r'''`tuple`(`DiscMeas`): The row counts for the observed data.
168        If `row` is $m - 1$ and `state` is $\ell$, then
169        `rowCounts`[`row`].`pmf`(`state`) is denoted by
170        $\overline{y}_{m\ell}$ in Section 9 of the Paper. It is the
171        number of observed instances of `state` in the given row of
172        data.
173        '''
174        return tuple(meas for meas in self._rowCounts)
